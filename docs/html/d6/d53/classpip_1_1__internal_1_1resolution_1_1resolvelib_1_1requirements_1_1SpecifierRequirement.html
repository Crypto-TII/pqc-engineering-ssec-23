66 def format_for_error(self) -> str:
67 # Convert comma-separated specifiers into "A, B, ..., F and G"
68 # This makes the specifier a bit more "human readable", without

70 # been checked)
71 parts = [s.strip() for s in str(self).split(",")]
72 if len(parts) == 0:

74 elif len(parts) == 1:
